🚀 ETAPA 3 — Criar/ajustar a VIEW de busca

from django.db.models import Q
from .forms import BuscaLivroForm
from .models import Livro  # precisa existir

def search_biblioteca(request):
    form = BuscaLivroForm(request.GET or None)
    livros = Livro.objects.all()

    if form.is_valid():
        query = form.cleaned_data.get('q')

        if query:
            livros = livros.filter(
                Q(nome__icontains=query) |
                Q(autor__icontains=query)
            )

    return render(
        request,
        'home/sections/minha_biblioteca.html',
        {
            'livros': livros,
            'form': form
        }
    )
